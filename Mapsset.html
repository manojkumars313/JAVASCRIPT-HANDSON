1."Problem: You are given a string (STR) of length N, consisting of only the lower case English alphabet. Your task is to remove all the duplicate occurrences of characters in the string. Input: abcadeecfb Output: abcdef"

            let output = "";
            
            for (let i = 0; i < input.length; i++) {
