const countCharacters = (S) =>

var Split_the_string = (S) => 
                                            {
                                              let separator = " ";
                                            let substrings = [];
                                            let currentSubstring = "";
                                            
                                            for (let i = 0; i < S.length; i++) {
                                                if (S[i] === separator) {
                                                    substrings.push(currentSubstring);
                                                    currentSubstring = "";
                                                } else {
                                                    currentSubstring += S[i];
                                                }
                                            }
                                            if (currentSubstring !== "") {
                                                substrings.push(currentSubstring);
                                            }
                                              
                                             return substrings;
                                            };
